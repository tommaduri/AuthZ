Thank you for installing {{ .Chart.Name }}!

Your release is named {{ .Release.Name }}.

To get started with authz-engine:

1. Get the application URL:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "authz-engine.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch the status by running:

        kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "authz-engine.fullname" . }}

  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "authz-engine.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.service.ports.http }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "authz-engine.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")

  echo "Visit http://127.0.0.1:8081 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 8081:$CONTAINER_PORT
{{- end }}

2. Check the health endpoint:
  kubectl run curl-test --rm -it --restart=Never --image=curlimages/curl -- \
    curl http://{{ include "authz-engine.fullname" . }}.{{ .Release.Namespace }}.svc.cluster.local:{{ .Values.service.ports.http }}/health

3. View the logs:
  kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "authz-engine.name" . }} -f

4. Check pod status:
  kubectl get pods -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "authz-engine.name" . }}

{{- if .Values.autoscaling.enabled }}

5. Monitor autoscaling:
  kubectl get hpa -n {{ .Release.Namespace }} {{ include "authz-engine.fullname" . }} -w
{{- end }}

{{- if .Values.serviceMonitor.enabled }}

6. View metrics in Prometheus:
  The ServiceMonitor has been created for Prometheus Operator.
  Metrics are available at: http://{{ include "authz-engine.fullname" . }}.{{ .Release.Namespace }}.svc.cluster.local:{{ .Values.service.ports.metrics }}/metrics
{{- end }}
